## 풀이1 정렬하여 딕셔너리에 추가

```py title="group_anagrams.py" showLineNumbers
def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
	anagrams = collections.defaultdict(list)
	
	for word in strs:
		anagrams[''.join(sorted(word))].append(word)
		
	return list(anagrams.values())
```

```js title="groupAnagrams.js" showLineNumbers
/**
 * @param {string[]} strs
 * @return {string[][]}
 */
var groupAnagrams = function(strs) {
    const setStrs = new Map();

    for(let i = 0; i < strs.length; i++) {
        let storage = strs[i].split('').sort().join('');

        if(setStrs.has(storage)) {
            setStrs.set(storage,[...setStrs.get(storage), strs[i]]); // 문자열 합치는 ... operator
        } else {
          setStrs.set(storage,[strs[i]]); // 세팅해주기
        }
    }

    let anagrams = [];
    setStrs.forEach((a, b) => {
      anagrams.push(a);
    })

    return anagrams;
};
```
